void LoginWindow::updateMusicIcon(bool muted)
{
    QString iconPath = muted ? ":/res/musicBimg2.jpg" : ":/res/musicBimg1.jpg";

    QPixmap pixmap(iconPath);
    if (pixmap.isNull()) {
        // 如果图片加载失败，使用备用方案（绘制简单图标）
        return;
    }

    // 将图片裁剪成圆形，更好地融入UI
    QPixmap rounded(pixmap.size());
    rounded.fill(Qt::transparent);

    QPainter painter(&rounded);
    painter.setRenderHint(QPainter::Antialiasing);
    painter.setBrush(QBrush(pixmap.scaled(pixmap.size(), Qt::IgnoreAspectRatio, Qt::SmoothTransformation)));
    painter.setPen(Qt::NoPen);
    painter.drawEllipse(rounded.rect());
    painter.end();

    musicBtn->setIcon(QIcon(rounded));
    musicBtn->setIconSize(QSize(28, 28));  // 图标比按钮略小，留出边距
}

void LoginWindow::showToast(const QString &message)
{
    if (!toastLabel) {
        toastLabel = new QLabel(nullptr);  // 独立窗口，无父对象
        toastLabel->setWindowFlags(Qt::ToolTip | Qt::FramelessWindowHint);
        // 重要：不要设置 WA_TranslucentBackground，否则背景和边框无法正常绘制
        toastLabel->setAlignment(Qt::AlignCenter);
        toastLabel->setStyleSheet(
            "QLabel {"
            "   background-color: rgb(62, 43, 31);"
            "   color: white;"
            "   border: 2px solid #9b7b62;"
            "   border-radius: 10px;"
            "   padding: 10px 20px;"
            "   font: bold 16px 'Microsoft YaHei', 'Arial', sans-serif;"
            "}"
            );
        toastLabel->hide();
    }

    toastLabel->setText(message);
    toastLabel->adjustSize();

    // 计算全局坐标：以父窗口为基准，水平居中，初始位于父窗口上方
    QPoint parentTopLeft = mapToGlobal(QPoint(0, 0));
    int winWidth = width();
    int toastW = toastLabel->width();
    int toastH = toastLabel->height();

    int globalStartX = parentTopLeft.x() + (winWidth - toastW) / 2;
    int globalStartY = parentTopLeft.y() - toastH;          // 父窗口上方
    int globalEndY = parentTopLeft.y() + 20;                 // 父窗口内距离顶部20像素

    toastLabel->move(globalStartX, globalStartY);
    toastLabel->show();
    toastLabel->raise();

    // 下降动画
    QPropertyAnimation *dropAnim = new QPropertyAnimation(toastLabel, "pos");
    dropAnim->setDuration(500);
    dropAnim->setStartValue(QPoint(globalStartX, globalStartY));
    dropAnim->setEndValue(QPoint(globalStartX, globalEndY));
    dropAnim->setEasingCurve(QEasingCurve::OutQuad);
    dropAnim->start(QPropertyAnimation::DeleteWhenStopped);

    // 停留1秒后上升消失
    QTimer::singleShot(1000, [this, globalStartX, globalEndY, toastH]() {
        if (toastLabel && toastLabel->isVisible()) {
            QPropertyAnimation *riseAnim = new QPropertyAnimation(toastLabel, "pos");
            riseAnim->setDuration(300);
            riseAnim->setStartValue(QPoint(globalStartX, globalEndY));
            riseAnim->setEndValue(QPoint(globalStartX, globalEndY - toastH - 20)); // 上升至父窗口上方
            riseAnim->setEasingCurve(QEasingCurve::InQuad);
            connect(riseAnim, &QPropertyAnimation::finished, toastLabel, &QLabel::hide);
            riseAnim->start(QPropertyAnimation::DeleteWhenStopped);
        }
    });
}

void LoginWindow::onLoginClicked()
{
    playSound("click");

    QString account = usernameEdit->text().trimmed();
    QString password = passwordEdit->text();

    if (account.isEmpty() || password.isEmpty()) {
        showToast("账号和密码不能为空");
        return;
    }

    if (!DBManager::userExists(account)) {
        showToast("账号不存在");
        return;
    }

    if (!DBManager::validateUser(account, password)) {
        showToast("密码错误");
        return;
    }

    // 登录成功，判断信息是否完善
    if (DBManager::hasUserProfile(account)) {
        // 已完善，直接跳转游戏主页
        showToast("登录成功！");
        QGraphicsOpacityEffect *opacityEffect = new QGraphicsOpacityEffect(this);
        this->setGraphicsEffect(opacityEffect);
        QPropertyAnimation *fadeOut = new QPropertyAnimation(opacityEffect, "opacity");
        fadeOut->setDuration(500);
        fadeOut->setStartValue(1.0);
        fadeOut->setEndValue(0.0);
        fadeOut->start(QPropertyAnimation::DeleteWhenStopped);

        QTimer::singleShot(1500, [this, account]() {
            emit showGameMainWindow(account);
            this->close();
        });
    } else {
        // 未完善，跳转完善信息
        showToast("请完善信息");
        QGraphicsOpacityEffect *opacityEffect = new QGraphicsOpacityEffect(this);
        this->setGraphicsEffect(opacityEffect);
        QPropertyAnimation *fadeOut = new QPropertyAnimation(opacityEffect, "opacity");
        fadeOut->setDuration(500);
        fadeOut->setStartValue(1.0);
        fadeOut->setEndValue(0.0);
        fadeOut->start(QPropertyAnimation::DeleteWhenStopped);

        QTimer::singleShot(1500, [this, account]() {
            emit showProfileWindow(account);
            this->close();
        });
    }
}

void LoginWindow::onRegisterClicked()
{
    playSound("click");
    emit showRegisterWindow();
    this->hide();
}

void LoginWindow::playSound(const QString &effect)
{
    // 只处理点击音效
    if (effect != "click")
        return;

    static QSoundEffect clickSound;
    static bool initialized = false;

    if (!initialized) {
        clickSound.setSource(QUrl("qrc:/res/btnSound.wav"));
        clickSound.setVolume(1.0f);
        initialized = true;
    }

    clickSound.play();
}
#ifndef MUSICMANAGER_H
#define MUSICMANAGER_H

#include <QObject>
#include <QMediaPlayer>
#include <QAudioOutput>

class MusicManager : public QObject
{
    Q_OBJECT
public:
    static MusicManager& instance();

    bool isMuted() const { return m_muted; }
    void setMuted(bool muted);   // true = 静音，false = 播放
    void toggleMute();            // 切换静音状态

signals:
    void muteStateChanged(bool muted);  // 状态变化信号，用于更新按钮图标

private:
    explicit MusicManager(QObject *parent = nullptr);
    ~MusicManager();

    QMediaPlayer *m_player;
    QAudioOutput *m_audioOutput;
    bool m_muted;
};

#endif // MUSICMANAGER_H
#include "musicmanager.h"
#include <QUrl>

MusicManager::MusicManager(QObject *parent)
    : QObject(parent), m_muted(false)
{
    m_player = new QMediaPlayer(this);
    m_audioOutput = new QAudioOutput(this);
    m_player->setAudioOutput(m_audioOutput);
    m_player->setSource(QUrl("qrc:/res/gameMB1.mp3")); // 背景音乐文件路径（请替换为您的文件）
    m_player->setLoops(QMediaPlayer::Infinite);    // 无限循环
    m_audioOutput->setVolume(0.5f);                 // 音量 0.0~1.0
    m_player->play();                               // 默认开始播放
}

MusicManager::~MusicManager()
{
    m_player->stop();
}

MusicManager& MusicManager::instance()
{
    static MusicManager instance;
    return instance;
}

void MusicManager::setMuted(bool muted)
{
    if (m_muted == muted) return;
    m_muted = muted;
    m_audioOutput->setMuted(m_muted);
    emit muteStateChanged(m_muted);
}

void MusicManager::toggleMute()
{
    setMuted(!m_muted);
}

void ProfileWindow::showToast(const QString &message, bool isSuccess)
{

        toastLabel->setAlignment(Qt::AlignCenter);
        toastLabel->setStyleSheet(
            "QLabel {"
            "   background-color: rgba(62, 43, 31, 220);"

            "   border-radius: 10px;"
            "   padding: 10px 20px;"
            "   font-size: 16px;"
            "   font-weight: bold;"
            "}"
            );
        toastLabel->hide();
    }

    toastLabel->setText(message);
    toastLabel->adjustSize();

    int winWidth = qMax(width(), 1);
    int toastW = qMax(toastLabel->width(), 1);
    int toastH = qMax(toastLabel->height(), 1);

    int startX = (winWidth - toastW) / 2;
    int startY = -toastH;
    int endY = 20;

    toastLabel->move(startX, startY);
    toastLabel->show();
    toastLabel->raise();

    QPropertyAnimation *dropAnim = new QPropertyAnimation(toastLabel, "pos");
    dropAnim->setDuration(300);
    dropAnim->setStartValue(QPoint(startX, startY));
    dropAnim->setEndValue(QPoint(startX, endY));
    dropAnim->setEasingCurve(QEasingCurve::OutQuad);
    dropAnim->start(QPropertyAnimation::DeleteWhenStopped);

        if (toastLabel && toastLabel->isVisible()) {
            QPropertyAnimation *riseAnim = new QPropertyAnimation(toastLabel, "pos");
            riseAnim->setDuration(300);
            riseAnim->setStartValue(toastLabel->pos());
            riseAnim->setEndValue(QPoint(toastLabel->x(), -toastLabel->height()));
            riseAnim->setEasingCurve(QEasingCurve::InQuad);
            connect(riseAnim, &QPropertyAnimation::finished, toastLabel, &QLabel::hide);
            riseAnim->start(QPropertyAnimation::DeleteWhenStopped);
        }

void ProfileWindow::onSelectAvatar()
{
    playSound("click");
    QString filePath = QFileDialog::getOpenFileName(this, "选择头像",
                                                    QDir::homePath(),
                                                    "图片文件 (*.png *.jpg *.jpeg *.bmp *.gif)");
    if (filePath.isEmpty())
        return;

    avatarPath = filePath;
    QPixmap pix(filePath);
    if (!pix.isNull()) {
        pix = pix.scaled(avatarLabel->size(), Qt::KeepAspectRatio, Qt::SmoothTransformation);
        avatarLabel->setPixmap(pix);
        avatarLabel->setText("");
    } else {
        avatarLabel->setText("无效图片");
    }
}

void ProfileWindow::onCompleteClicked()
{
    playSound("click");

    QString nickname = nicknameEdit->text().trimmed();
    if (nickname.isEmpty() || nickname.length() > 25) {
        showToast("昵称必须为1~25个字符");
        return;
    }

    QString gender = maleRadio->isChecked() ? "男" : "女";

    if (DBManager::saveUserProfile(m_account, nickname, gender, avatarPath)) {
        QString message = QString("侦探：%1的冒险开始").arg(nickname);
        showToast(message);
        // 延迟跳转到游戏主页，确保提示完整显示
        QTimer::singleShot(1800, [this]() {
            emit goToGameMain(m_account);
            this->close();

void RegisterWindow::updateMusicIcon(bool muted)
{
    QString iconPath = muted ? ":/res/musicBimg2.jpg" : ":/res/musicBimg1.jpg";

    QPixmap pixmap(iconPath);
    if (pixmap.isNull()) {
        qDebug() << "音乐图标加载失败:" << iconPath;
        return;
    }

    // 将图片裁剪成圆形
    QPixmap rounded(pixmap.size());
    rounded.fill(Qt::transparent);

    QPainter painter(&rounded);
    painter.setRenderHint(QPainter::Antialiasing);
    painter.setBrush(QBrush(pixmap.scaled(pixmap.size(), Qt::IgnoreAspectRatio, Qt::SmoothTransformation)));
    painter.setPen(Qt::NoPen);
    painter.drawEllipse(rounded.rect());
    painter.end();

    musicBtn->setIcon(QIcon(rounded));
    musicBtn->setIconSize(QSize(28, 28));  // 图标比按钮略小，留出边距
}

bool RegisterWindow::validatePassword(const QString &pwd)
{
    if (pwd.length() < 6 || pwd.length() > 10) return false;
    bool hasLower = false, hasUpper = false;
    for (QChar ch : pwd) {
        if (ch.isLower()) hasLower = true;
        if (ch.isUpper()) hasUpper = true;
    }
    return hasLower && hasUpper;
}

void RegisterWindow::showToast(const QString &message)
{
    if (!toastLabel) {
        // 创建独立窗口，不受父窗口透明度影响
        toastLabel = new QLabel(nullptr);
        toastLabel->setWindowFlags(Qt::ToolTip | Qt::FramelessWindowHint);
        toastLabel->setAlignment(Qt::AlignCenter);
        toastLabel->setStyleSheet(
            "QLabel {"
            "   background-color: rgb(62, 43, 31);"
            "   color: white;"
            "   border: 2px solid #9b7b62;"
            "   border-radius: 10px;"
            "   padding: 10px 20px;"
            "   font: bold 16px 'Microsoft YaHei', 'Arial', sans-serif;"
            "}"
            );
        toastLabel->hide();
    }

    toastLabel->setText(message);
    toastLabel->adjustSize();

    QPoint parentTopLeft = mapToGlobal(QPoint(0, 0));
    int winWidth = width();
    int toastW = toastLabel->width();
    int toastH = toastLabel->height();

    int globalStartX = parentTopLeft.x() + (winWidth - toastW) / 2;
    int globalStartY = parentTopLeft.y() - toastH;
    int globalEndY = parentTopLeft.y() + 20;

    toastLabel->move(globalStartX, globalStartY);
    toastLabel->show();
    toastLabel->raise();

    QPropertyAnimation *dropAnim = new QPropertyAnimation(toastLabel, "pos");
    dropAnim->setDuration(500);
    dropAnim->setStartValue(QPoint(globalStartX, globalStartY));
    dropAnim->setEndValue(QPoint(globalStartX, globalEndY));
    dropAnim->setEasingCurve(QEasingCurve::OutQuad);
    dropAnim->start(QPropertyAnimation::DeleteWhenStopped);

    QTimer::singleShot(1000, [this, globalStartX, globalEndY, toastH]() {
        if (toastLabel && toastLabel->isVisible()) {
            QPropertyAnimation *riseAnim = new QPropertyAnimation(toastLabel, "pos");
            riseAnim->setDuration(300);
            riseAnim->setStartValue(QPoint(globalStartX, globalEndY));
            riseAnim->setEndValue(QPoint(globalStartX, globalEndY - toastH - 20));
            riseAnim->setEasingCurve(QEasingCurve::InQuad);
            connect(riseAnim, &QPropertyAnimation::finished, toastLabel, &QLabel::hide);
            riseAnim->start(QPropertyAnimation::DeleteWhenStopped);
        }
    });
}

void RegisterWindow::onConfirmClicked()
{
    playSound("click");

    QString account = usernameEdit->text().trimmed();
    QString password = passwordEdit->text();
    QString confirm = confirmEdit->text();

    if (account.length() != 11) {
        showToast("账号必须为11位");
        return;
    }

    if (!validatePassword(password)) {
        showToast("密码必须为6-20位，且包含大小写字母");
        return;
    }

    if (password != confirm) {
        showToast("两次输入的密码不一致");
        return;
    }

    if (DBManager::userExists(account)) {
        showToast("账号已存在，请直接登录");
        return;
    }

    if (DBManager::addUserCredential(account, password)) {
        showToast("注册完成！");
        QTimer::singleShot(1800, [this]() {
            onBackClicked();
        });
    } else {
        showToast("注册失败，请稍后重试");
    }
}

void RegisterWindow::onBackClicked()
{
    playSound("click");
    emit showLoginWindow();
    this->close();
}

        clickSound.setSource(QUrl("qrc:/res/btnSound.wav"));
        clickSound.setVolume(1.0f);
        initialized = true;
    }

    clickSound.play();
}

void LevelBase::showToast(const QString &message)
{
    QLabel *toast = new QLabel(message, nullptr);
    toast->setWindowFlags(Qt::ToolTip | Qt::FramelessWindowHint);
    toast->setAlignment(Qt::AlignCenter);
    toast->setStyleSheet(
        "QLabel {"
        "   background-color: rgb(62, 43, 31);"
        "   color: white;"
        "   border: 2px solid #9b7b62;"
        "   border-radius: 10px;"
        "   padding: 10px 20px;"
        "   font: bold 16px 'Microsoft YaHei', 'Arial', sans-serif;"
        "}"
        );
    toast->adjustSize();

    QPoint parentTopLeft = mapToGlobal(QPoint(0, 0));
    int winWidth = width();
    int toastW = toast->width();
    int toastH = toast->height();

    int globalStartX = parentTopLeft.x() + (winWidth - toastW) / 2;
    int globalStartY = parentTopLeft.y() - toastH;
    int globalEndY = parentTopLeft.y() + 20;

    toast->move(globalStartX, globalStartY);
    toast->show();
    toast->raise();

    QPropertyAnimation *dropAnim = new QPropertyAnimation(toast, "pos");
    dropAnim->setDuration(500);
    dropAnim->setStartValue(QPoint(globalStartX, globalStartY));
    dropAnim->setEndValue(QPoint(globalStartX, globalEndY));
    dropAnim->setEasingCurve(QEasingCurve::OutQuad);
    dropAnim->start(QPropertyAnimation::DeleteWhenStopped);

    QTimer::singleShot(1000, [toast, globalStartX, globalEndY, toastH]() {
        if (toast && toast->isVisible()) {
            QPropertyAnimation *riseAnim = new QPropertyAnimation(toast, "pos");
            riseAnim->setDuration(300);
            riseAnim->setStartValue(QPoint(globalStartX, globalEndY));
            riseAnim->setEndValue(QPoint(globalStartX, globalEndY - toastH - 20));
            riseAnim->setEasingCurve(QEasingCurve::InQuad);
            connect(riseAnim, &QPropertyAnimation::finished, toast, &QLabel::deleteLater);
            riseAnim->start(QPropertyAnimation::DeleteWhenStopped);
        }
    });
}

void LevelBase::onUploadClue()
{
    playClickSound();

    if (m_cellClueNames.size() >= 49) {
        qDebug() << "背包已满，无法添加新线索";
        return;
    }

    QString filePath = QFileDialog::getOpenFileName(this, "选择线索图片", QDir::homePath(), "图片文件 (*.png *.jpg *.jpeg *.bmp *.gif)");

    if (filePath.isEmpty()) {
        return;
    }

    showNamingDialog(filePath);
}

void LevelBase::onCaptureClue()
{
    playClickSound();

    if (m_cellClueNames.size() >= 49) {
        qDebug() << "背包已满，无法添加新线索";
        return;
    }

    QDialog *captureDialog = new QDialog(this);
    captureDialog->setWindowTitle("拍照线索");
    captureDialog->setFixedSize(640, 520);
    captureDialog->setStyleSheet(R"(
        QDialog {
            background-color: rgb(62, 43, 31);
            border: 3px solid #9b7b62;
            border-radius: 15px;
        }
        QPushButton {
            background-color: #7a5a44;
            border: 2px solid #9b7b62;
            border-radius: 10px;
            padding: 8px 16px;
            color: #e0d6c0;
            font-weight: bold;
            font-size: 14px;
        }
        QPushButton:hover {
            background-color: #9b7b62;
        }
        QPushButton:pressed {
            background-color: #5c3f2e;
        }
    )");

    QCamera *camera = new QCamera(QMediaDevices::defaultVideoInput(), captureDialog);
    QVideoWidget *viewfinder = new QVideoWidget(captureDialog);
    viewfinder->setFixedSize(600, 400);

    QMediaCaptureSession *captureSession = new QMediaCaptureSession(captureDialog);
    captureSession->setCamera(camera);
    captureSession->setVideoOutput(viewfinder);

    QImageCapture *imageCapture = new QImageCapture(captureDialog);
    imageCapture->setFileFormat(QImageCapture::JPEG);
    captureSession->setImageCapture(imageCapture);

    QVBoxLayout *layout = new QVBoxLayout(captureDialog);
    layout->addWidget(viewfinder);

    QHBoxLayout *btnLayout = new QHBoxLayout();
    QPushButton *captureBtn = new QPushButton("拍照", captureDialog);
    QPushButton *cancelBtn = new QPushButton("取消", captureDialog);
    btnLayout->addStretch();
    btnLayout->addWidget(captureBtn);
    btnLayout->addWidget(cancelBtn);
    btnLayout->addStretch();
    layout->addLayout(btnLayout);

    captureDialog->setLayout(layout);

    camera->start();

    connect(captureBtn, &QPushButton::clicked, this, [=]() {
        playClickSound();

        QString timestamp = QDateTime::currentDateTime().toString("yyyyMMdd_hhmmsszzz");
        QString tempDir = QCoreApplication::applicationDirPath() + "/MyGameClues";
        QDir dir(tempDir);
        if (!dir.exists()) dir.mkpath(".");
        QString filePath = tempDir + "/capture_" + timestamp + ".jpg";

        imageCapture->captureToFile(filePath);
    });

    connect(imageCapture, &QImageCapture::imageSaved, this, [=](int id, const QString &fileName) {
        Q_UNUSED(id);
        camera->stop();
        captureDialog->accept();
        showNamingDialog(fileName);
    });

    connect(cancelBtn, &QPushButton::clicked, captureDialog, &QDialog::reject);

    captureDialog->exec();
    captureDialog->deleteLater();
}

void LevelBase::addClueWithName(const QString &clueName, const QString &imagePath)
{
    int gridIndex = DBManager::getAvailableGridIndex(m_account);
    if (gridIndex == -1) {
        qDebug() << "背包已满";
        if (m_nameDialog) m_nameDialog->close();
        return;
    }

    if (DBManager::addBackpackClue(m_account, clueName, imagePath, gridIndex)) {
        updateCellWithClue(gridIndex, clueName, imagePath);
        qDebug() << "线索上传成功";
    } else {
        qDebug() << "上传失败，请重试";
    }

    if (m_nameDialog) m_nameDialog->close();
}

void LevelBase::showBackpackBox()
{
    if (!m_backpackBox) {
        createBackpackBox();
    }

    m_backpackOverlay->raise();
    m_backpackBox->raise();
    m_backpackOverlay->show();

    int boxWidth = m_backpackBox->width();
    int boxHeight = m_backpackBox->height();
    int startX = -boxWidth;
    int startY = (height() - boxHeight) / 2;
    int endX = 0;

    m_backpackBox->move(startX, startY);
    m_backpackBox->show();

    QPropertyAnimation *anim = new QPropertyAnimation(m_backpackBox, "pos");
    anim->setDuration(400);
    anim->setStartValue(QPoint(startX, startY));
    anim->setEndValue(QPoint(endX, startY));
    anim->setEasingCurve(QEasingCurve::OutQuad);
    anim->start(QPropertyAnimation::DeleteWhenStopped);
}

void LevelBase::hideBackpackBox()
{
    if (!m_backpackBox || !m_backpackBox->isVisible()) return;

    int startX = m_backpackBox->x();
    int startY = m_backpackBox->y();
    int endX = -m_backpackBox->width();

    QPropertyAnimation *anim = new QPropertyAnimation(m_backpackBox, "pos");
    anim->setDuration(300);
    anim->setStartValue(QPoint(startX, startY));
    anim->setEndValue(QPoint(endX, startY));
    anim->setEasingCurve(QEasingCurve::InQuad);

    connect(anim, &QPropertyAnimation::finished, [this]() {
        m_backpackBox->hide();
        if (m_backpackOverlay) m_backpackOverlay->hide();
    });

    anim->start(QPropertyAnimation::DeleteWhenStopped);
}

void LevelBase::loadBackpackClues()
{
    QList<BackpackClue> clues = DBManager::getBackpackClues(m_account);

    for (const BackpackClue &clue : qAsConst(clues)) {
        updateCellWithClue(clue.gridIndex, clue.clueName, clue.imagePath);
    }

    qDebug() << "已加载" << clues.size() << "个背包线索";
}

void LevelBase::updateCellWithClue(int gridIndex, const QString &clueName, const QString &imagePath)
{
    if (!m_backpackCells.contains(gridIndex)) return;

    QLabel *cell = m_backpackCells[gridIndex];

    QPixmap pixmap(imagePath);
    if (!pixmap.isNull()) {
        QPixmap scaled = pixmap.scaled(45, 45, Qt::KeepAspectRatio, Qt::SmoothTransformation);
        cell->setPixmap(scaled);
    }

    cell->setToolTip(clueName);

    m_cellClueNames[cell] = clueName;
    m_cellImagePaths[cell] = imagePath;
}

bool LevelBase::eventFilter(QObject *obj, QEvent *event)
{
    QLabel *cell = qobject_cast<QLabel*>(obj);
    if (cell && m_cellClueNames.contains(cell)) {
        if (event->type() == QEvent::MouseButtonDblClick) {
            QMouseEvent *mouseEvent = static_cast<QMouseEvent*>(event);
            if (mouseEvent->button() == Qt::LeftButton) {
                playClickSound();
                QString imagePath = m_cellImagePaths[cell];
                QString clueName = m_cellClueNames[cell];
                showEnlargedImage(imagePath, clueName);
                return true;
            }
        }
    }
    return QWidget::eventFilter(obj, event);
}

// ========== 根据关卡ID和类别ID返回线索名称 ==========
QString LevelBase::getClueNameByLevel(int levelId, int classId) const
{
    switch(levelId) {
    case 1:
        switch(classId) {
        case 0: return "狗";
        case 1: return "苹果";
        default: return "有用线索";
        }
    case 2:
        switch(classId) {
        case 0: return "鸟";
        case 1: return "风筝";
        case 2: return "香蕉";
        default: return "有用线索";
        }
    case 3:
        switch(classId) {
        case 0: return "刀";
        case 1: return "钟";
        case 2: return "花瓶";
        default: return "有用线索";
        }
    default:
        return "有用线索";
    }
}

// ========== YOLO 识别函数（最终工作版本） ==========
QString LevelBase::identifyClueWithResult(const cv::Mat &image)
{
    static QMap<int, cv::dnn::Net> modelCache;

    if (m_modelPath.isEmpty()) {
        qDebug() << "❌ 模型路径为空";
        return "";
    }
    if (!QFile::exists(m_modelPath)) {
        qDebug() << "❌ 模型不存在:" << m_modelPath;
        return "";
    }

    if (!modelCache.contains(m_levelId)) {
        cv::dnn::Net net = cv::dnn::readNetFromONNX(m_modelPath.toStdString());
        if (net.empty()) {
            qDebug() << "❌ 加载ONNX失败";
            return "";
        }
        net.setPreferableBackend(cv::dnn::DNN_BACKEND_OPENCV);
        net.setPreferableTarget(cv::dnn::DNN_TARGET_CPU);
        modelCache[m_levelId] = net;
        qDebug() << "✅ 关卡" << m_levelId << "模型加载成功";
    }

    cv::dnn::Net &net = modelCache[m_levelId];
    if (net.empty() || image.empty()) return "";

    cv::Mat blob = cv::dnn::blobFromImage(image, 1/255.0, cv::Size(640,640), cv::Scalar(), true, false);
    net.setInput(blob);
    cv::Mat out;
    net.forward(out);

    int numClasses = getClassCountByLevel(m_levelId);
    int nb = out.size[2];  // 8400

    float confThresh = 0.25f;
    float bestScore = 0;
    int bestCls = -1;

    const float *data = (float*)out.data;

    // 使用您原来正确的索引方式（类别优先）
    for (int cls = 0; cls < numClasses; cls++) {
        for (int i = 0; i < nb; i++) {
            float score = data[(4 + cls) * nb + i];
            if (score > confThresh && score > bestScore) {
                bestScore = score;
                bestCls = cls;
            }
        }
    }

    qDebug() << "关卡" << m_levelId << "最佳识别 - 类别:" << bestCls << "置信度:" << bestScore;

    if (bestCls >= 0 && bestCls < numClasses) {
        QString name = getClueNameByLevel(m_levelId, bestCls);
        qDebug() << "识别结果：" << name;
        return name;
    }

    return "";
}

bool LevelBase::identifyClue(const cv::Mat &image)
{
    if(image.empty()) {
        qDebug() << "图像为空，无法识别";
        return false;
    }
    QString result = identifyClueWithResult(image);
    bool success = !result.isEmpty();
    if (success) {
        qDebug() << "识别成功，线索名称:" << result;
    } else {
        qDebug() << "识别失败，未找到匹配的线索";
    }
    return success;
}

bool LevelBase::isTargetClue(const QString &clueName) const
{
    switch(m_levelId) {
    case 1:
        return (clueName == "狗" || clueName == "苹果");
    case 2:
        return (clueName == "鸟" || clueName == "风筝" || clueName == "香蕉");
    case 3:
        return (clueName == "刀" || clueName == "钟" || clueName == "花瓶");
    default:
        return false;
    }
}

void LevelBase::updateCameraFrame()
{
    if (!m_cap.isOpened() || !m_isCameraRunning) {
        return;
    }

    m_cap >> m_currentFrame;
    if (m_currentFrame.empty()) {
        return;
    }

    QString clueType = identifyClueWithResult(m_currentFrame);

    if (!clueType.isEmpty()) {
        cv::rectangle(m_currentFrame, cv::Rect(50, 50, 200, 200), cv::Scalar(0, 255, 0), 3);
        cv::putText(m_currentFrame, clueType.toStdString(),
                    cv::Point(60, 40), cv::FONT_HERSHEY_SIMPLEX,
                    0.8, cv::Scalar(0, 255, 0), 2);

        if (m_cameraDialog && m_cameraViewLabel) {
            QMetaObject::invokeMethod(this, [this, clueType]() {
                if (m_cameraDialog) {
                    QList<QLabel*> labels = m_cameraDialog->findChildren<QLabel*>();
                    for (QLabel *label : labels) {
                        if (label->text() == "等待识别..." || label->text().startsWith("识别到:")) {
                            label->setText(QString("识别到: %1").arg(clueType));
                            label->setStyleSheet("color: #d4af37; font-size: 14px; font-weight: bold;");
                            break;
                        }
                    }
                }
            });
        }

        static bool alreadyAdded = false;
        if (!alreadyAdded && isTargetClue(clueType)) {
            alreadyAdded = true;

            QString timestamp = QDateTime::currentDateTime().toString("yyyyMMdd_hhmmsszzz");
            QString tempDir = QCoreApplication::applicationDirPath() + "/MyGameClues";
            QDir dir(tempDir);
            if (!dir.exists()) dir.mkpath(".");
            QString filePath = tempDir + "/identified_" + timestamp + ".jpg";
            cv::imwrite(filePath.toStdString(), m_currentFrame);

            showNamingDialogForIdentify(filePath, clueType);

            QTimer::singleShot(3000, []() { alreadyAdded = false; });
        }
    } else {
        if (m_cameraDialog) {
            QMetaObject::invokeMethod(this, [this]() {
                if (m_cameraDialog) {
                    QList<QLabel*> labels = m_cameraDialog->findChildren<QLabel*>();
                    for (QLabel *label : labels) {
                        if (label->text().startsWith("识别到:")) {
                            label->setText("等待识别...");
                            label->setStyleSheet("color: #e0d6c0; font-size: 14px;");
                            break;
                        }
                    }
                }
            });
        }
    }

    cv::Mat displayFrame;
    cv::cvtColor(m_currentFrame, displayFrame, cv::COLOR_BGR2RGB);
    QImage img(displayFrame.data, displayFrame.cols, displayFrame.rows,
               displayFrame.step, QImage::Format_RGB888);
    QPixmap pixmap = QPixmap::fromImage(img.scaled(m_cameraViewLabel->size(),
                                                   Qt::KeepAspectRatio,
                                                   Qt::SmoothTransformation));
    if (m_cameraViewLabel) {
        m_cameraViewLabel->setPixmap(pixmap);
    }
}

void LevelBase::captureAndIdentify()
{
    if (!m_cap.isOpened() || m_currentFrame.empty()) {
        qDebug() << "无法捕获当前帧";
        return;
    }

    playClickSound();

    cv::Mat frameCopy = m_currentFrame.clone();

    bool isUseful = identifyClue(frameCopy);

    if (isUseful) {
        QString timestamp = QDateTime::currentDateTime().toString("yyyyMMdd_hhmmsszzz");
        QString tempDir = QCoreApplication::applicationDirPath() + "/MyGameClues";
        QDir dir(tempDir);
        if (!dir.exists()) dir.mkpath(".");
        QString filePath = tempDir + "/identified_" + timestamp + ".jpg";

        cv::imwrite(filePath.toStdString(), frameCopy);

        QString clueType = identifyClueWithResult(frameCopy);
        showNamingDialogForIdentify(filePath, clueType);

        stopCamera();
        if (m_cameraDialog) {
            m_cameraDialog->accept();
        }
    } else {
        showIdentifyResult(false, "");
    }
}

bool LevelBase::addToUsefulClueBox(const QString &clueType, const QString &clueName, const QString &imagePath)
{
    qDebug() << "添加有用线索 - 类型:" << clueType << "名称:" << clueName;

    for (int i = 0; i < m_usefulClueIndex; i++) {
        if (m_usefulClueCells.contains(i)) {
            QLabel *existingCell = m_usefulClueCells[i];
            QString existingType = existingCell->property("clueType").toString();
            if (existingType == clueType) {
                showToast(QString("已有同类线索「%1」，不能重复添加！").arg(clueType));
                qDebug() << "物体类型" << clueType << "已经存在，不重复添加";
                return false;
            }
        }
    }

    if (m_usefulClueIndex >= 9) {
        showToast("有用线索框已满");
        qDebug() << "有用线索框已满";
        return false;
    }

    if (m_usefulClueCells.contains(m_usefulClueIndex)) {
        QLabel *cell = m_usefulClueCells[m_usefulClueIndex];

        QPixmap pixmap(imagePath);
        if (!pixmap.isNull()) {
            QPixmap scaled = pixmap.scaled(45, 45, Qt::KeepAspectRatio, Qt::SmoothTransformation);
            cell->setPixmap(scaled);
        }

        cell->setToolTip(clueName);
        cell->setProperty("clueType", clueType);
        m_usefulClueIndex++;

        addProgress(1);
        showToast(QString("✓ 获得有用线索：%1").arg(clueName));
        qDebug() << "有用线索添加成功，当前进度:" << m_usefulCluesCount;
        return true;
    }

    return false;
}

void LevelBase::showNotebookBox()
{
    if (!m_notebookBox) {
        createNotebookBox();
    }

    m_isNotebookOpen = true;
    setInteractionEnabled(false);

    m_notebookOverlay->raise();
    m_notebookBox->raise();
    m_notebookOverlay->show();

    int boxWidth = m_notebookBox->width();
    int boxHeight = m_notebookBox->height();
    int startX = width();
    int startY = (height() - boxHeight) / 2;
    int endX = width() - boxWidth - 10;

    m_notebookBox->move(startX, startY);
    m_notebookBox->show();

    QPropertyAnimation *anim = new QPropertyAnimation(m_notebookBox, "pos");
    anim->setDuration(400);
    anim->setStartValue(QPoint(startX, startY));
    anim->setEndValue(QPoint(endX, startY));
    anim->setEasingCurve(QEasingCurve::OutQuad);
    connect(anim, &QPropertyAnimation::finished, this, [this]() {
        m_notebookText->setFocus();
    });
    anim->start(QPropertyAnimation::DeleteWhenStopped);
}

void LevelBase::hideNotebookBox()
{
    if (!m_notebookBox || !m_notebookBox->isVisible()) return;

    int startX = m_notebookBox->x();
    int startY = m_notebookBox->y();
    int endX = width();

    QPropertyAnimation *anim = new QPropertyAnimation(m_notebookBox, "pos");
    anim->setDuration(300);
    anim->setStartValue(QPoint(startX, startY));
    anim->setEndValue(QPoint(endX, startY));
    anim->setEasingCurve(QEasingCurve::InQuad);

    connect(anim, &QPropertyAnimation::finished, [this]() {
        m_notebookBox->hide();
        if (m_notebookOverlay) m_notebookOverlay->hide();
        m_isNotebookOpen = false;
        setInteractionEnabled(true);
    });

    anim->start(QPropertyAnimation::DeleteWhenStopped);
}

void LevelBase::hideCharacterDialog()
{
    if (m_typewriterTimer) {
        m_typewriterTimer->stop();
        delete m_typewriterTimer;
        m_typewriterTimer = nullptr;
    }

    if (m_dialogWidget) {
        m_dialogWidget->hide();
        m_dialogWidget->deleteLater();
        m_dialogWidget = nullptr;
    }

    m_avatarLabel = nullptr;
    m_nameLabel = nullptr;
    m_dialogText = nullptr;
    m_dialogContinueBtn = nullptr;
    m_isDialogShowing = false;

    setDialogInteractionEnabled(true);
}

void LevelBase::startCurrentDialog()
{
    if (m_currentDialogIndex >= m_currentDialogues.size()) {
        hideCharacterDialog();
        return;
    }

    m_currentTargetText = m_currentDialogues[m_currentDialogIndex];
    m_currentDisplayText = "";
    m_typewriterPos = 0;

    m_dialogText->clear();

    if (m_typewriterTimer) {
        m_typewriterTimer->stop();
    }
    m_typewriterTimer->start(50);
}

void LevelBase::updateDialogText()
{
    if (m_typewriterPos < m_currentTargetText.length()) {
        m_currentDisplayText += m_currentTargetText.at(m_typewriterPos);
        m_dialogText->setPlainText(m_currentDisplayText);

        QTextCursor cursor = m_dialogText->textCursor();
        cursor.movePosition(QTextCursor::End);
        m_dialogText->setTextCursor(cursor);

        m_typewriterPos++;
    } else {
        m_typewriterTimer->stop();

        if (m_currentDialogIndex < m_currentDialogues.size() - 1) {
            m_dialogContinueBtn->setText("继续");
        } else {
            m_dialogContinueBtn->setText("完成");
        }
    }
}

void LevelBase::onDialogContinue()
{
    playClickSound();

    if (m_typewriterTimer && m_typewriterTimer->isActive()) {
        m_typewriterTimer->stop();
        m_dialogText->setPlainText(m_currentTargetText);
        m_typewriterPos = m_currentTargetText.length();

        if (m_currentDialogIndex < m_currentDialogues.size() - 1) {
            m_dialogContinueBtn->setText("继续");
        } else {
            m_dialogContinueBtn->setText("完成");
        }
        return;
    }

    m_currentDialogIndex++;
    startCurrentDialog();
}

        }

            background-color: #5c3f2e;
        }

    subLabel->setAlignment(Qt::AlignCenter);
    subLabel->setStyleSheet("font-size: 14px; color: #e0d6c0;");
    layout->addWidget(subLabel);

    QPushButton *backBtn = new QPushButton("返回主页");

    backBtn->setFixedWidth(150);  // 设置固定宽度，更好看

    // 将按钮添加到布局中，居中显示
    layout->addWidget(backBtn, 0, Qt::AlignCenter);

    // 连接返回按钮
    connect(backBtn, &QPushButton::clicked, this, [this, completeDialog, completeOverlay]() {
        playClickSound();

        // 先标记关卡完成（但不自动进入下一关）
        emit levelCompleted(m_levelId);

        // 关闭弹窗
        completeDialog->accept();
        completeOverlay->deleteLater();

        // 延迟关闭当前关卡窗口，返回主页
        QTimer::singleShot(200, this, [this]() {
            emit backToMain();
        });
    });

    // 强制居中
    completeDialog->move(
        (this->width() - completeDialog->width()) / 2,
        (this->height() - completeDialog->height()) / 2
        );

    // 确保显示在最上层
    completeDialog->show();
    completeDialog->raise();

    // 弹窗关闭时删除遮罩
    connect(completeDialog, &QDialog::finished, completeOverlay, &QWidget::deleteLater);
}
